#include <iostream>
#include <vector>
#include <iomanip>
#include <algorithm>

using namespace std;

const int MAX_SUM = 10001;
int dp[101][MAX_SUM];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;

    vector<int> a(n), b(n);
    int b_total = 0;
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> b[i];
        b_total += b[i];
    }

    // Initialize DP with a very small number
    for (int i = 0; i <= n; i++) {
        for (int j = 0; j < MAX_SUM; j++) {
            dp[i][j] = -1e9;
        }
    }
    dp[0][0] = 0;

    for (int k = 0; k < n; k++) {
        for (int i = k; i >= 0; i--) {
            for (int j = 0; j <= 100 * i; j++) {
                if (dp[i][j] < 0) continue;
                dp[i + 1][j + a[k]] = max(dp[i + 1][j + a[k]], dp[i][j] + b[k]);
            }
        }
    }

    cout << fixed << setprecision(10);
    for (int k = 1; k <= n; k++) {
        double max_water = 0;
        for (int cap = 0; cap < MAX_SUM; cap++) {
            if (dp[k][cap] < 0) continue;
            // The formula: min(Capacity, (Total_B + Subset_B) / 2)
            double current = min((double)cap, (b_total + dp[k][cap]) / 2.0);
            max_water = max(max_water, current);
        }
        cout << max_water << (k == n ? "" : " ");
    }
    cout << endl;

    return 0;
}
