#include <iostream>
#include <vector>
#include <string>

using namespace std;

vector<string> tokens;
string result;
int pos = 0;
bool possible = true;

void parse() {
    if (pos >= tokens.size()) {
        possible = false;
        return;
    }

    string current = tokens[pos++];
    if (current == "pair") {
        result += "pair<";
        parse();
        result += ",";
        parse();
        result += ">";
    } else {
        result += "int";
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;

    string s;
    while (cin >> s) {
        tokens.push_back(s);
    }

    parse();

    if (!possible || pos != tokens.size()) {
        cout << "Error occurred" << endl;
    } else {
        cout << result << endl;
    }

    return 0;
}
