class Solution:
    def maxRunTime(self, n: int, batteries: List[int]) -> int:
        l=1
        r=sum(batteries)//n
        batteries = sorted(batteries)
        res=0
        while l<=r:
            mid = int((l+r)/2)
            if self.valid_time(mid,n,batteries):
                res=mid
                l=mid+1
            else:
                r=mid-1
        return res
    def valid_time(self,mid,n,batteries):
        x=n*mid
        temp=0
        for i in range(len(batteries)):
            temp+=min(batteries[i],mid)
        if temp >=x:
            return True
        return False
